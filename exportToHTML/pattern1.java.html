import java.util.Scanner;
public class pattern1 {
    public static void pattern(int n)
    {
        for(int i=0;i<n;i++)
        {
            for(int j=0;j<n;j++)
            {
                System.out.print("-");
            }
            System.out.print("\n");
        }

    }

    public static void main(String[] args) {
        int n;
        Scanner s = new Scanner(System.in);
        n = s.nextInt();
        pattern(n);
    }
}
